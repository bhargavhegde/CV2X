48  0x02, 0x04, 0x00, 0x01, 0xE2, 0x40, 0x00, 0x08,

58  /* Decode UPER encoded SPaT to C struct. This effectively allocates
59  a EU_SPAT instance and fills it by the buffer contents. */
60  if(0 >= asn1_uper_decode((void**)&spat,
61  asn1_type_EU_SPAT,
62  ORIGINAL_SPAT,
63  sizeof(ORIGINAL_SPAT),
64  &decode_err)) {
65  printf("Size: %d", (int)sizeof(ORIGINAL_SPAT));

71  }

73  /* Modify the SPaT message, in this case add a 3. MovementEvent to 3. MovementState */
74  error = error || modify_spat(spat);

76  if(!error) {

80  if(FALSE == asn1_check_constraints(asn1_type_EU_SPAT, spat, &err)) {
