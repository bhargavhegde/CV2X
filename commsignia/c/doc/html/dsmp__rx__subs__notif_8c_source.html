59 {

61 
62  const char* host = (argc > 1) ? argv[1] : "127.0.0.1";
63 
64  /* Create a session */
65  cms_session_t session = cms_get_session();
66 
67  /* Connect to the host */
68  if(cms_api_connect_easy(&session, host)) {
69  printf("Unable to connect\n");
70  error = true;
71  }
72 
73  /* Subscribe to DSMP rx notification (filter to AID) */
74  cms_subs_id_t subs_id_filter = 0; /* Output of the subscribe function */
75 
76  notif_ctx_t callback_ctx_filter = {
77  .param = CONTEXT_FILTER_AID,
78  .rx_cnt = 0
79  };
80 
81  static const cms_psid_t AID = 0x11;
82 
83  error = error || cms_dsmp_rx_subscribe(&session,
84  AID,
85  dsmp_rx_notif_cb,
86  &callback_ctx_filter,
87  &subs_id_filter);
88 
89  /* Subscribe to DSMP rx notification (do not filter to AID) */
90  cms_subs_id_t subs_id_all = 0; /* Output of the subscribe function */
91 
92  notif_ctx_t wildcard_ctx = {
93  .param = CONTEXT_ALL_AID,
94  .rx_cnt = 0
95  };
96 
97  error = error || cms_dsmp_rx_subscribe(&session,
98  CMS_DSMP_SUBSCRIBE_WILDCARD, /* subscribe to all AID */
99  dsmp_rx_notif_cb,
100  &wildcard_ctx,
101  &subs_id_all);
102 
103  if(!error) {
104  /* Wait some messages on the wildcard subscription */
105  static const uint32_t EXIT_ON_TX_COUNT = 50UL;
106  while(wildcard_ctx.rx_cnt < EXIT_ON_TX_COUNT) {
107  sleep(1);
108  }
109  } else {
110  printf("Unable to subscribe to all DSMP notifications\n");
111  }
112 
113  /* Unsubscribe */
114  error = error || cms_dsmp_rx_unsubscribe(&session, subs_id_filter);
115  error = error || cms_dsmp_rx_unsubscribe(&session, subs_id_all);
116 
117  /* Close connection and cleanup */
118  cms_api_disconnect(&session);
119  cms_api_clean();
120 
121  return (int) error;
122 }
