163 static bool process_uper(cms_fac_msg_type_t type, cms_buffer_view_t msg)
164 {
165  bool error = false;
166  void* c_struct = NULL;
167  const ASN1CType* asn_type = get_type(type);
168  if(NULL == asn_type) {
169  error = true;
170  printf("Unable to detect asn type\n");
171  }
172 
173  /* decode uper buffer to c struct */
174  if(!error) {
175  fprintf(stderr, "Decoding %d bytes\n", msg.length);
176  ASN1Error err;
177  int ret = asn1_uper_decode((void**)&c_struct, asn_type, msg.data, msg.length, &err);
178  if((ret < 0) || (c_struct == NULL)) {
179  fprintf(stderr, "Decoding error: %s\n", err.msg);
180  error = true;
181  }
182  }
183 
184  /* use c struct according to your needs */
185  /* in this example we will print number of intersections in EU & US MAP messages */
186  error = error || l_use_c_stuct(type, c_struct);
187 
188  /* encode c struct to human readable xml struct and print to stdout */
189  error = error || l_print_xml_format(asn_type, c_struct);
190 
191  if(c_struct != NULL) {
192  asn1_free_value(asn_type, c_struct);
193  }
194 
195  return error;
196 }

198 static bool l_use_c_stuct(cms_fac_msg_type_t type, void* c_struct)
199 {
200  bool error = false;
201 
202  switch(type) {
203  case CMS_FAC_MSG_EU_MAP: {
204  EU_MAP* c_map = (EU_MAP*)c_struct;
205  if(c_map->map.intersections_option) {
206  printf("number of intersections: %zu\n", c_map->map.intersections.count);
207  }
208  break;
209  }
210  case CMS_FAC_MSG_US_MAP: {
211  US_MessageFrame* c_message_frame = (US_MessageFrame*)c_struct;
212  if(c_message_frame->value.type != asn1_type_US_MapData) {
213  printf("Error, Message is not us map");
214  error = true;
215  } else {
216  US_MapData* map_data = (US_MapData*)c_message_frame->value.u.data;
217  if(map_data->intersections_option) {
218  printf("number of intersections: %zu\n", map_data->intersections.count);
219  }
220  }
221  break;
222  }
223  default:
224  /* modify message handling according to your needs */
225  break;
226  }
227 
228  return error;
229 }
